git status
    verificar mudanças no projeto

git add
    adicionar novos arquivos para o git
    git add css/
    git add .

git commit 
    salvar alterações do projeto
    git commit estudo.txt -m "enviando o txt"
    git commit -a -m "mudancas 11_05_26"

git push 
    enviar codigo para repositorio remoto
    git push origin main
    git push

git pull
    sincronizar repo local com o repo remoto
    git pull origin main

git clone
    clonar um repositorio remoto (baixar um repositorio remoto)
    git clone https://github.com/joaopaulofaria/repo_estudos.git .

git rm
    remover arquivos da monitoração do git
    git rm c.txt
    #depois de usar o rm é preciso comitar e push

git log
    log de modificacoes no projeto
    git log

git mv
    renomear arquivo ou mover para outra pasta
    git mv b.css c.css
    git mv rodape.css css/rodape.css

git checkout
    retornar arquivo modificado ao estado original (estado do repositorio)
    git checkout css/styles.css

.gitignore
    ignorar arquivos com informações sensiveis do projeto

git reset
    resetar as mudanças feitas na branch (todas as alterações commitadas e pendentes)
    git reset --hard origin/main

branch
    ramificaçõs do projeto
    separa versoes do projeto

git branch
    cria e lista branchs
    git branch primeiro_branch
    -d (para deletar branch)

git checkout nome_branch
    muda de branch
    -b para criar e mudar

*comitar e salvar todas as alterações antes de mudar de branch*
*criar uma branch b estando na a cria uma nova branch b a partir da a

git merge
    unir branches
    git merge primeiro_branch

*se eu estou na main e rodo git merche primeiro_branch, o conteudo de primeiro_branch é juntado com o conteudo da main

git stash
    salvar as modificações atuais para prosseguuir com outra abordagem de solução
     git stash
     git stash list
        verifica as stashs criadas
    git stash apply 0
        recuperar arquivos da stash
    git stash clear
    git stash drop name_stash

git tag
    tag é um checkpoint de uma branch (demarcar estagios de desenvolvimento)
    git tag -a nome_tag -m "mensagem"
    git tag
        lista tags
    git show nome_tag
    git checkout nome_tag
        trocar de tag
    git push origin nome_tag
        enviar tag para o repo remoto

git fetch
    lista todas as branchs e tags de outro dev (que nao estao reconhecidas)
    git fetch -a

git remote
    remover e adicionar origens
    git remote -v
    git remote rm origin
    git remote add origin_url


submodulo
    Possuir dois ou mais projetos em um repo
    git submodule add url_repo
    git push --recurse-submodules=on-demand
        atualizar submodule

git show
    informações uteis do branch atual, tags e dos commits
    git show

git diff
    exibe as diferenças de uma branch, arquivo
    git diff primeiro_branch
    git diff arquivo1 arquivo2

git shortlog 
    log resumido do projeto
    git shortlog

git clean
    verificar e limpar arquivos nao trackeados (que nao foi usado o git add ainda)
    git clean
